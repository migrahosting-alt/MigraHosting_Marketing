```typescript
async function checkDomain(domain: string): Promise<boolean> {
  const API_KEY = import.meta.env.VITE_NAMECHEAP_API_KEY;
  const response = await fetch(`https://api.namecheap.com/xml.response`, {
    params: {
      ApiUser: import.meta.env.VITE_NAMECHEAP_USERNAME,
      ApiKey: API_KEY,
      Command: 'namecheap.domains.check',
      DomainList: domain,
    },
  });
  return parseXML(response).available;
}
```
